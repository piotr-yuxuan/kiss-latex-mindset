\usepackage{hyperref}
\hypersetup{
	bookmarks=true,         % show bookmarks bar?
	breaklinks=true,
	unicode=true,          % non-Latin characters in Acrobat’s bookmarks
	pdftoolbar=true,        % show Acrobat’s toolbar?
	pdfmenubar=true,        % show Acrobat’s menu?
	pdffitwindow=false,     % window fit to page when opened
	pdfstartview={FitH},    % fits the width of the page to the window
	pdftitle={\customtitle{}},    % title
	pdfauthor={\customauthor{}},     % author
	pdfsubject={\customsubject{}},   % subject of the document
	pdfcreator={\customcreator{}},   % creator of the document
	pdfproducer={\customproducer{}}, % producer of the document
	pdfkeywords={\customkeywords{}}, % list of keywords
	pdfnewwindow=true,      % links in new PDF window
	colorlinks=false,       % false: boxed links; true: colored links
	linkcolor=red,          % color of internal links (change box color with linkbordercolor)
	citecolor=green,        % color of links to bibliography
	filecolor=magenta,      % color of file links
	urlcolor=cyan           % color of external links
}
